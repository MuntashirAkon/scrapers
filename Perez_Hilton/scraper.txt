function perez_set_category_retargeting_cookie(post_categories, site_version) {
    if (typeof (post_categories) == "undefined") {
        return;
    }

    var cookie_name = ph_retargeting_cookie_name;
    var cookie_value = "";
    var cookie_value_array;
    var cookie_value_current = $.cookie(cookie_name);
    var cookie_value_max_length = 70;
    var needed_categories = ph_retargeting_needed_categories;

    for (var i = 0; i < perez_object_length(needed_categories); i++) {
        if ($.inArray(needed_categories[i]["cat_id"], post_categories) >= 0 &&
                (!cookie_value_current || cookie_value_current.toLowerCase().indexOf("|" + needed_categories[i]["cat_id"] + ",") < 0)) {
            cookie_value += "|" + needed_categories[i]["cat_id"] + ",";

            if (typeof (perez_statLog) == "function") {
                perez_statLog({
                    app: 'perez_category_retargeting',
                    options: 'd',
                    category: perez_retargeting_replace_cat_name(needed_categories[i]["name"]) + "-" + site_version
                });
            }

            perez_retargeting_call_tradedesk(needed_categories[i]["td_id"]);
        }
    }

    if (cookie_value !== "") {
        if (cookie_value_current) {
            cookie_value = cookie_value + cookie_value_current;
        }

        cookie_value_array = cookie_value.split(",");

        if (typeof (cookie_value_array) == "object" && cookie_value_array.length > cookie_value_max_length) {
            cookie_value_array = cookie_value_array.slice(0, cookie_value_max_length);
            cookie_value = cookie_value_array.join();
            cookie_value = cookie_value + ",";
        }

        perez_setCookie(cookie_name, cookie_value, 730);
    }
}

function perez_retargeting_replace_cat_name(cat_name) {
    cat_name = cat_name.replace(/ /gi, '_');
    cat_name = cat_name.replace(/\./gi, '_');
    cat_name = cat_name.replace(/\'/gi, '_');
    cat_name = cat_name.replace(/\"/gi, '_');

    return cat_name;
}

function perez_retargeting_call_tradedesk(td_id) {

        $("body").prepend('<div id="td_id_' + td_id + '" style="top: 0px; left: 0px; width: 1px; height: 1px; border: none; position: absolute;"></div>');

}

function perez_retargeting_get_ga_custom_var() {
    var cookie_value = $.cookie(ph_retargeting_cookie_name);

    if (typeof (cookie_value) == "undefined" ||
            cookie_value === "" ||
            cookie_value === null ||
            typeof (_gaq) !== 'object') {
        return;
    }

    var categories = ph_retargeting_needed_categories;
    var item;
    var custom_var = "";
    var cookie_value_array = cookie_value.split(",");
    var current_cats = 0;
    var max_cats = 42;

    for (var i = 0; i < cookie_value_array.length; i++) {
        if (cookie_value_array[i] !== "" && current_cats < max_cats) {
            item = cookie_value_array[i];
            item = item.replace(/\|/gi, '').replace(/,/gi, '');
            item = parseInt(item);

            for (var j = 0; j < perez_object_length(categories); j++) {
                if (categories[j]["cat_id"] == item && categories[j]["ga_top_celeb"] === true) {
                    custom_var += categories[j]["ga_id"] + ",";
                    current_cats = current_cats + 1;
                    break;
                }
            }
        }
    }

    if (custom_var !== "") {
        _gaq.push(['_setCustomVar', 1, "cp", custom_var, 1]);
    }
}
